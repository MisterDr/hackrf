201 void DataRam_WriteBlocks(USB_ClassInfo_MS_Device_t *const MSInterfaceInfo,
202  const uint32_t BlockAddress,
203  uint16_t TotalBlocks)
204 {
205  uint32_t i;
206 
207  if (!InitializedFlag) {
208  DataRam_Initialize();
209  }
210 
211  for (i = 0; i < TotalBlocks; i++) {
212  while (!Endpoint_IsReadWriteAllowed(MSInterfaceInfo->Config.PortNumber)) {}
213 
214  // Fake the read if the address is past the physical limit of the DiskImage buffer
215  if (BlockAddress + i >= DATA_RAM_PHYSICAL_SIZE / VIRTUAL_MEMORY_BLOCK_SIZE) {
216  Endpoint_Null_Stream(MSInterfaceInfo->Config.PortNumber, VIRTUAL_MEMORY_BLOCK_SIZE, NULL);
217  Endpoint_ClearOUT(MSInterfaceInfo->Config.PortNumber);
218  continue;
219  }
220 
221  Endpoint_Read_Stream_LE(MSInterfaceInfo->Config.PortNumber,
222  &DiskImage[(BlockAddress + i) * VIRTUAL_MEMORY_BLOCK_SIZE],
223  VIRTUAL_MEMORY_BLOCK_SIZE,
224  NULL);
225  Endpoint_ClearOUT(MSInterfaceInfo->Config.PortNumber);
226  }
227 }

263 uint32_t MassStorage_GetAddressInImage(uint32_t startblock, uint16_t requestblocks, uint16_t *availableblocks)
264 {
265  if ((startblock + requestblocks) <= DATA_RAM_PHYSICAL_SIZE / DATA_RAM_BLOCK_SIZE) {
266  *availableblocks = requestblocks;
267  }
268  else {
269  if (startblock >= DATA_RAM_PHYSICAL_SIZE / DATA_RAM_BLOCK_SIZE) {
270  *availableblocks = 0;
271  }
272  else {*availableblocks = DATA_RAM_PHYSICAL_SIZE / DATA_RAM_BLOCK_SIZE - startblock; }
273  }
274 
275  if (*availableblocks != 0) {
276  return (uint32_t) &DiskImage[startblock * DATA_RAM_BLOCK_SIZE];
277  }
278  else {
279  return (uint32_t) &DiskImage[(DATA_RAM_PHYSICAL_SIZE / DATA_RAM_BLOCK_SIZE - 1) * DATA_RAM_BLOCK_SIZE];
