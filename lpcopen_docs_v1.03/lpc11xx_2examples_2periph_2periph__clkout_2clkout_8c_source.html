94 int main(void)
95 {
96  CHIP_SYSCTL_CLKOUTSRC_T clkoutClks;
97 
98  Board_Init();
99 
100  /* Enable and setup SysTick Timer at a 100Hz rate */
101  SysTick_Config(SystemCoreClock / 100);
102 
103  /* Enable the power to the WDT */
104  Chip_SYSCTL_PowerUp(SYSCTL_POWERDOWN_WDTOSC_PD);
105 
106  /*This example uses the watchdog oscillator set at a 50KHz (1Mhz / 20) clock rate. */
107  Chip_Clock_SetWDTOSC(WDTLFO_OSC_1_05, 20);
108 
109 #if defined(CHIP_LPC11UXX) || defined(CHIP_LPC11EXX) || defined(CHIP_LPC11AXX)
110  /* LPC11U14 Xpresso board has CLKOUT on pin PIO0_1 on J6-38 */
111  Chip_IOCON_PinMuxSet(LPC_IOCON, 0, 1, (IOCON_FUNC1 | IOCON_MODE_INACT));
112 #else
113  /* LPC11C24 Xpresso board has CLKOUT on pin PIO0_1 on J6-38 */
114  Chip_IOCON_PinMuxSet(LPC_IOCON, IOCON_PIO0_1, (IOCON_FUNC1 | IOCON_MODE_INACT));
115 #endif
116 
117  /* Cycle through all clock sources for the CLKOUT pin */
118  for (clkoutClks = SYSCTL_CLKOUTSRC_IRC;
119  clkoutClks <= SYSCTL_CLKOUTSRC_MAINSYSCLK; clkoutClks++) {
120  ticks100 = 0;
121 
122  /* Setup CLKOUT pin for specific clock with a divider of 1 */
123  Chip_Clock_SetCLKOUTSource(clkoutClks, 1);
124 
125  /* Wait 5 seconds */
126  while (ticks100 < 500) {
127  __WFI();
128  }
129  }
130 
131  /* Disable CLKOUT pin by setting divider to 0 */
132  Chip_Clock_SetCLKOUTSource(SYSCTL_CLKOUTSRC_IRC, 0);
133 
134  return 0;
135 }
