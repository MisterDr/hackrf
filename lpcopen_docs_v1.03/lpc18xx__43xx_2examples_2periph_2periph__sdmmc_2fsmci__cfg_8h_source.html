139 STATIC INLINE int FSMCI_CardReadyWait(CARD_HANDLE_T *hCrd, int tout)
140 {
141  int32_t curr = (int32_t) Chip_RIT_GetCounter(LPC_RITIMER);
142  int32_t final = curr + ((SystemCoreClock / 1000) * tout);
143 
144  if ((final < 0) && (curr > 0)) {
145  while (Chip_RIT_GetCounter(LPC_RITIMER) < (uint32_t) final) { if (Chip_SDMMC_GetState(LPC_SDMMC) != -1) break; }
146  }
147  else {
148  while ((int32_t) Chip_RIT_GetCounter(LPC_RITIMER) < final) { if (Chip_SDMMC_GetState(LPC_SDMMC) != -1) break; }
149  }
150 
151  return Chip_SDMMC_GetState(LPC_SDMMC) != -1;
152 }
