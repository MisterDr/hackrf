113 int main(void)
114 {
115  uint32_t timerFreq;
116 
117  Board_Init();
118 
119  DEBUGSTR("Blinky example using timers 1 and 2!\r\n");
120 
121  /* Enable timer 1 clock and get clock rate */
122  Chip_TIMER_Init(LPC_TIMER1);
123 #if defined(CHIP_LPC175X_6X)
124  timerFreq = Chip_Clock_GetPeripheralClockRate(SYSCTL_PCLK_TIMER1);
125 #else
126  timerFreq = Chip_Clock_GetPeripheralClockRate();
127 #endif
128 
129  Chip_TIMER_Reset(LPC_TIMER1);
130  Chip_TIMER_MatchEnableInt(LPC_TIMER1, 1);
131  Chip_TIMER_SetMatch(LPC_TIMER1, 1, (timerFreq / TICKRATE1_HZ));
132  Chip_TIMER_ResetOnMatchEnable(LPC_TIMER1, 1);
133  Chip_TIMER_Enable(LPC_TIMER1);
134 
135  /* Enable timer interrupt */
136  NVIC_EnableIRQ(TIMER1_IRQn);
137  NVIC_ClearPendingIRQ(TIMER1_IRQn);
138 
139  DEBUGOUT("Timer 1 clock = %d Hz\r\n", timerFreq);
140  DEBUGOUT("Timer 1 tick rate = %d Hz\r\n", TICKRATE1_HZ);
141 
142  /* Enable timer 2 clock and get clock rate */
143  Chip_TIMER_Init(LPC_TIMER2);
144 #if defined(CHIP_LPC175X_6X)
145  timerFreq = Chip_Clock_GetPeripheralClockRate(SYSCTL_PCLK_TIMER2);
146 #else
147  timerFreq = Chip_Clock_GetPeripheralClockRate();
148 #endif
149 
150  Chip_TIMER_Reset(LPC_TIMER2);
151  Chip_TIMER_MatchEnableInt(LPC_TIMER2, 1);
152  Chip_TIMER_SetMatch(LPC_TIMER2, 1, (timerFreq / TICKRATE2_HZ));
153  Chip_TIMER_ResetOnMatchEnable(LPC_TIMER2, 1);
154  Chip_TIMER_Enable(LPC_TIMER2);
155 
156  /* Enable timer interrupt */
157  NVIC_EnableIRQ(TIMER2_IRQn);
158  NVIC_ClearPendingIRQ(TIMER2_IRQn);
159 
160  DEBUGOUT("Timer 2 clock = %d Hz\r\n", timerFreq);
161  DEBUGOUT("Timer 2 tick rate = %d Hz\r\n", TICKRATE2_HZ);
162 
163  while (1) {
164  __WFI();
165  }
166 }
