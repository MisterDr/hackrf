103 int main(void)
104 {
105  uint32_t timerFreq;
106 
107  Board_Init();
108 
109  /* Enable and setup SysTick Timer at a periodic rate */
110  SysTick_Config(SystemCoreClock / TICKRATE_HZ1);
111 
112  /* Enable timer 1 clock */
113  Chip_TIMER_Init(LPC_TIMER32_0);
114 
115  /* Timer rate is system clock rate */
116  timerFreq = Chip_Clock_GetSystemClockRate();
117 
118  /* Timer setup for match and interrupt at TICKRATE_HZ */
119  Chip_TIMER_Reset(LPC_TIMER32_0);
120  Chip_TIMER_MatchEnableInt(LPC_TIMER32_0, 1);
121  Chip_TIMER_SetMatch(LPC_TIMER32_0, 1, (timerFreq / TICKRATE_HZ2));
122  Chip_TIMER_ResetOnMatchEnable(LPC_TIMER32_0, 1);
123  Chip_TIMER_Enable(LPC_TIMER32_0);
124 
125  /* Enable timer interrupt */
126  NVIC_ClearPendingIRQ(TIMER_32_0_IRQn);
127  NVIC_EnableIRQ(TIMER_32_0_IRQn);
128 
129  /* LEDs toggle in interrupt handlers */
130  while (1) {
131  __WFI();
132  }
133 
134  return 0;
135 }
