50 /* Sets up the CRC engine with defaults based on the polynomial to be used */
51 void IP_CRC_UseDefaultConfig(IP_CRC_001_T *pCRC, IP_CRC_001_POLY_T poly)
52 {
53  switch (poly) {
54  case CRC_POLY_CRC16:
55  pCRC->MODE = MODE_CFG_CRC16;
56  pCRC->SEED = CRC_SEED_CRC16;
57  break;
58 
59  case CRC_POLY_CRC32:
60  pCRC->MODE = MODE_CFG_CRC32;
61  pCRC->SEED = CRC_SEED_CRC32;
62  break;
63 
64  case CRC_POLY_CCITT:
65  default:
66  pCRC->MODE = MODE_CFG_CCITT;
67  pCRC->SEED = CRC_SEED_CCITT;
68  break;
69  }
70 }

72 /* Convenience function for computing a standard CCITT checksum from an 8-bit data block */
73 uint32_t IP_CRC_CRC8(IP_CRC_001_T *pCRC, const uint8_t *data, uint32_t bytes)
74 {
75  while (bytes > 0) {
76  IP_CRC_Write8(pCRC, *data);
77  data++;
78  bytes--;
79  }
80  return IP_CRC_ReadSum(pCRC);
81 }
