108 int main(void)
109 {
110  bool On = false;
111 
112  Board_Init();
113 
114  /* Init Alarm Timer with Preset Count for about 1s */
115  Chip_ATIMER_Init(LPC_ATIMER, PresetCount);
116 
117  /* Init EVRT */
118  Chip_EVRT_Init();
119 
120  /* Enable EVRT in order to be able to read the ATIMER interrupt */
121  Chip_EVRT_ConfigIntSrcActiveType(EVRT_SRC_ATIMER, EVRT_SRC_ACTIVE_HIGH_LEVEL);
122 
123  /* Enable Alarm Timer Source */
124  Chip_EVRT_SetUpIntSrc(EVRT_SRC_ATIMER, ENABLE);
125 
126  /* Enable NVIC */
127  NVIC_EnableIRQ(EVENTROUTER_IRQn);
128 
129  /* Clear the interrupt states */
130  ATIMER_ClearInts();
131 
132  /* Enable Alarm Timer */
133  Chip_ATIMER_IntEnable(LPC_ATIMER);
134 
135  while (1) {
136  /* Sleep until ATIMER fires */
137  __WFI();
138  On = (bool) !On;
139  Board_LED_Set(1, On);
140  }
141 }
