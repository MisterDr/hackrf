50 /* Generate a palette table (only in 8-bit mode) */
51 void lpc_colors_set_palette(uint16_t *palette_table)
52 {
53 #if COLORS_DEF == 8
54  int32_t idx;
55  uint16_t entry, r, g, b;
56 
57  /* 256 entries */
58  for (idx = 0; idx < NUM_COLORS; idx++) {
59  r = ((uint16_t) idx & REDMASK) >> REDSHIFT;
60  g = ((uint16_t) idx & GREENMASK) >> GREENSHIFT;
61  b = ((uint16_t) idx & BLUEMASK) >> BLUESHIFT;
62 
63 #ifdef COLORS_8_565_MODE
64  /* Strip out and scale colors */
65  r = r * 0x1F / ((REDMASK >> REDSHIFT) + 1);
66  g = g * 0x3F / ((GREENMASK >> GREENSHIFT) + 1);
67  b = b * 0x1F / ((BLUEMASK >> BLUESHIFT) + 1);
68  entry = b + (g << 5) + (r << 11);
69 
70 #else
71  /* Strip out and scale colors */
72  r = r * 0x1F / ((REDMASK >> REDSHIFT) + 1);
73  g = g * 0x1F / ((GREENMASK >> GREENSHIFT) + 1);
74  b = b * 0x1F / ((BLUEMASK >> BLUESHIFT) + 1);
75  entry = b + (g << 5) + (r << 10);
76 #endif
77 
78  /* Save palette entry */
79  palette_table[idx] = entry;
80  }
81 #endif /* COLORS_DEF == 8 */
82 }
