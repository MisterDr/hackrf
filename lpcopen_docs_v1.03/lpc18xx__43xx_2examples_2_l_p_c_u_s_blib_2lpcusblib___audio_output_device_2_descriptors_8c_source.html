294 uint16_t CALLBACK_USB_GetDescriptor(uint8_t corenum,
295  const uint16_t wValue,
296  const uint8_t wIndex,
297  const void * *const DescriptorAddress)
298 {
299  const uint8_t DescriptorType = (wValue >> 8);
300  const uint8_t DescriptorNumber = (wValue & 0xFF);
301 
302  const void *Address = NULL;
303  uint16_t Size = NO_DESCRIPTOR;
304 
305  switch (DescriptorType) {
306  case DTYPE_Device:
307  Address = &DeviceDescriptor;
308  Size = sizeof(USB_Descriptor_Device_t);
309  break;
310 
311  case DTYPE_Configuration:
312  Address = &ConfigurationDescriptor;
313  Size = sizeof(USB_Descriptor_Configuration_t);
314  break;
315 
316  case DTYPE_String:
317  switch (DescriptorNumber) {
318  case 0x00:
319  Address = LanguageStringPtr;
320  Size = pgm_read_byte(&LanguageStringPtr->Header.Size);
321  break;
322 
323  case 0x01:
324  Address = ManufacturerStringPtr;
325  Size = pgm_read_byte(&ManufacturerStringPtr->Header.Size);
326  break;
327 
328  case 0x02:
329  Address = ProductStringPtr;
330  Size = pgm_read_byte(&ProductStringPtr->Header.Size);
331  break;
332  }
333 
334  break;
335  }
336 
337  *DescriptorAddress = Address;
338  return Size;
339 }
