131 /* OS specific event hanlder for FreeRTOS */
132 static void os_event_handler(void)
133 {
134  portBASE_TYPE wake1 = pdFALSE, wake2 = pdFALSE;
135 
136  if (event_rx && !QUEUE_IS_EMPTY(qrd)) {
137  xSemaphoreGiveFromISR(event_rx, &wake1);
138  }
139 
140  if (event_tx && !QUEUE_IS_FULL(qwr)) {
141  xSemaphoreGiveFromISR(event_tx, &wake2);
142  }
143 
144  portEND_SWITCHING_ISR(wake1 || wake2);
145 }

261 /* Function to initialize the IPC message queue */
262 void IPC_initMsgQueue(void *data, int size, int count)
263 {
264  /* Sanity Check */
265  if (!size || !count || !data) {
266  DEBUGSTR("ERROR:IPC Queue size invalid parameters\r\n");
267  while (1) {}
268  }
269 
270  /* Check if size is a power of 2 */
271  if (count & (count - 1)) {
272  DEBUGSTR("ERROR:IPC Queue size not power of 2\r\n");
273  while (1) { /* BUG: Size must always be power of 2 */
274  }
275  }
276 
277  memset(qwr, 0, sizeof(*qwr));
278  ipc_misc_init();
279  qwr->count = count;
280  qwr->size = size;
281  qwr->data = data;
282  qwr->valid = QUEUE_MAGIC_VALID;
283  NVIC_SetPriority(IPC_IRQn, IPC_IRQ_Priority);
284  NVIC_EnableIRQ(IPC_IRQn);
285 }

287 /* Function to push a message into queue with timeout */
288 int IPC_pushMsgTout(const void *data, int tout)
289 {
290  /* Check if write queue is initialized */
291  if (!QUEUE_IS_VALID(qwr)) {
292  return QUEUE_ERROR;
293  }
294 
295  if (tout == 0) {
296  /* Check if queue is full */
297  if (QUEUE_IS_FULL(qwr)) {
298  return QUEUE_FULL;
299  }
300  }
301  else if (tout < 0) {
302  /* Wait for read queue to have some data */
303  ipc_wait_event(QUEUE_IS_FULL(qwr), event_tx);
304  }
305  else {
306  /* Wait for read queue to have some data */
307  ipc_wait_event_tout(QUEUE_IS_FULL(qwr), tout, event_tx);
308  if (tout == 0) {
309  return QUEUE_TIMEOUT;
310  }
311  }
312 
313  memcpy(qwr->data + ((qwr->head & (qwr->count - 1)) * qwr->size), data, qwr->size);
314  qwr->head++;
315  ipc_send_signal();
316 
317  return QUEUE_INSERT;
318 }

320 /* Function to read a message from queue with timeout */
321 int IPC_popMsgTout(void *data, int tout)
322 {
323 #ifdef EVENT_ON_RX
324  int raise_event = QUEUE_IS_FULL(qrd);
325 #endif
326 
327  if (!QUEUE_IS_VALID(qrd)) {
328  return QUEUE_ERROR;
329  }
330 
331  if (tout == 0) {
332  /* Check if read queue is empty */
333  if (QUEUE_IS_EMPTY(qrd)) {
334  return QUEUE_EMPTY;
335  }
336  }
337  else if (tout < 0) {
338  /* Wait for read queue to have some data */
339  ipc_wait_event(QUEUE_IS_EMPTY(qrd), event_rx);
340  }
341  else {
342  /* Wait for event or timeout */
343  ipc_wait_event_tout(QUEUE_IS_EMPTY(qrd), tout, event_rx);
344  if (tout == 0) {
345  return QUEUE_TIMEOUT;
346  }
347  }
348 
349  /* Pop the queue Item */
350  memcpy(data, qrd->data + ((qrd->tail & (qrd->count - 1)) * qrd->size), qrd->size);
351  qrd->tail++;
352 
353 #ifdef EVENT_ON_RX
354  if (raise_event) {
355  ipc_send_signal();
356  }
357 #endif
358  return QUEUE_VALID;
359 }
