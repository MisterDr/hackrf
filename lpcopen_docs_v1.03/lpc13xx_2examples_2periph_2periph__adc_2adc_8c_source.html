83  char out_str[5];
84  for (j = 4; j != 0; j--) {
85  out_str[j - 1] = (data % 10) + 48;
86  data /= 10;
87  }
88  out_str[4] = '\0';
89 
90  DEBUGSTR("\r\nADC value is ");

92 
93  /* Delay */
94  j = 500000;

96 }

98 /* Simple ADC resr and display loop */
99 static void App_Polling_Test(void)
100 {
101  uint16_t dataADC;
102 
103  while (1) {
104  /* Start A/D conversion */
105  Chip_ADC_Set_StartMode(_LPC_ADC_ID, ADC_START_NOW, ADC_TRIGGERMODE_RISING);
106 
107  /* Waiting for A/D conversion complete */
108  while (Chip_ADC_Read_Status(_LPC_ADC_ID, ADC_CH0, ADC_DR_DONE_STAT) != SET) {}
109 
110  /* Read ADC value */
111  Chip_ADC_Read_Value(_LPC_ADC_ID, ADC_CH0, &dataADC);
112 
113  /* Print ADC value */
114  App_print_ADC_value(dataADC);
115  }
116 }
