50 /* Initialize SPI peripheral block */
51 void IP_SPI_Init(IP_SPI_002_T *pSPI, IP_SPI_CONFIG_T *pConfig)
52 {
53  uint32_t EnStat = pSPI->CFG & SPI_CFG_SPI_EN;
54 
55  /* Disable before update CFG register */
56  if (EnStat) {
57  IP_SPI_Disable(pSPI);
58  }
59 
60  /* SPI Configurate */
61  pSPI->CFG = ((uint32_t) pConfig->ClockMode) | ((uint32_t) pConfig->DataOrder) | ((uint32_t) pConfig->Mode) |
62  ((uint32_t) pConfig->SSELPol);
63 
64  /* Rate Divider setting */
65  pSPI->DIV = SPI_DIV_VAL(pConfig->ClkDiv);
66 
67  /* Clear status flag*/
68  IP_SPI_ClearStatus(pSPI, SPI_STAT_CLR_RXOV | SPI_STAT_CLR_TXUR | SPI_STAT_CLR_SSA | SPI_STAT_CLR_SSD);
69 
70  /* Return the previous state */
71  if (EnStat) {
72  IP_SPI_Enable(pSPI);
73  }
74 }

76 /* Configure SPI Delay parameters */
77 void IP_SPI_DelayConfig(IP_SPI_002_T *pSPI, IP_SPI_DELAY_CONFIG_T *pConfig)
78 {
79  pSPI->DLY = SPI_DLY_PRE_DELAY(pConfig->PreDelay);
80  pSPI->DLY |= SPI_DLY_POST_DELAY(pConfig->PostDelay);
81  pSPI->DLY |= SPI_DLY_FRAME_DELAY(pConfig->FrameDelay);
82  if (pConfig->TransferDelay) {
83  pSPI->DLY |= SPI_DLY_TRANSFER_DELAY(pConfig->TransferDelay - 1);
84  }
85 }
