123 
124  /* The match/capture REGMODE defaults to match mode */
125 
126  /* Set the match count for match register 0 */
127  Chip_SCT_SetMatchCount(LPC_SCT, SCT_MATCH_0, SystemCoreClock / TICKRATE_HZ);
128 
129  /* Set the match reload value */

131 
132  /* Enable an Interrupt on the Match Event */
