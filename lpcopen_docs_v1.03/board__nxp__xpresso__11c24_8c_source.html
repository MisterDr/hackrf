62 /* Sends a character on the UART */
63 void Board_UARTPutChar(char ch)
64 {
65 #if defined(DEBUG_ENABLE)
66  /* Wait for space in FIFO */
67  while ((IP_UART_ReadLineStatus(DEBUG_UART) & UART_LSR_THRE) == 0);
68  Chip_UART_SendByte(DEBUG_UART, (uint8_t) ch);
69 #endif
70 }

72 /* Gets a character from the UART, returns EOF if no character is ready */
73 int Board_UARTGetChar(void)
74 {
75 #if defined(DEBUG_ENABLE)
76  if (Chip_UART_ReadLineStatus(DEBUG_UART) & UART_LSR_RDR) {
77  return (int) Chip_UART_ReadByte(DEBUG_UART);
78  }
79 #endif
80  return EOF;
81 }

100  Chip_UART_Init(LPC_USART);

102  Chip_UART_ConfigData(LPC_USART, (UART_LCR_WLEN8 | UART_LCR_SBS_1BIT));
