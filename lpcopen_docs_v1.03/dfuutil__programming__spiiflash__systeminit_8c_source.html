77 void dfuUtilSystemInit(void)
78 {
79  int i;
80  unsigned int *pSCB_VTOR = (unsigned int *) 0xE000ED08;
81 
82 #if defined(__IAR_SYSTEMS_ICC__)
83  extern void *__vector_table;
84 
85  *pSCB_VTOR = (unsigned int) &__vector_table;
86 #elif defined(__CODE_RED)
87  extern void *g_pfnVectors;
88 
89  *pSCB_VTOR = (unsigned int) &g_pfnVectors;
90 #elif defined(__ARMCC_VERSION)
91  extern void *__Vectors;
92 
93  *pSCB_VTOR = (unsigned int) &__Vectors;
94 #endif
95 
96  /* SPIFI pin setup is done prior to setting up system clocking */
97  for (i = 0; i < (sizeof(spifipinmuxing) / sizeof(spifipinmuxing[0])); i++) {
98  Chip_SCU_PinMuxSet(spifipinmuxing[i].pingrp, spifipinmuxing[i].pinnum,
99  spifipinmuxing[i].modefunc);
100  }
101 }
