117 int main(void)

130  /* Static IP assignment */
131 #if LWIP_DHCP
132  IP4_ADDR(&gw, 0, 0, 0, 0);
133  IP4_ADDR(&ipaddr, 0, 0, 0, 0);
134  IP4_ADDR(&netmask, 0, 0, 0, 0);
135 #else
136  IP4_ADDR(&gw, 10, 1, 10, 1);
137  IP4_ADDR(&ipaddr, 10, 1, 10, 234);

139 // APP_PRINT_IP(&ipaddr);
140 #endif
141 
142  /* Add netif interface for lpc17xx_8x */
143  netif_add(&lpc_netif, &ipaddr, &netmask, &gw, NULL, lpc_enetif_init,
144  ethernet_input);
145  netif_set_default(&lpc_netif);
146  netif_set_up(&lpc_netif);
147 
148 #if LWIP_DHCP
149  dhcp_start(&lpc_netif);
150 #endif
151 
152  /* Initialize and start application */
153  echo_init();
154 
155 
156  iperf_server_init();
157 
158  /* This could be done in the sysTick ISR, but may stay in IRQ context
159  too long, so do this stuff with a background loop. */
160  while (1) {
161  /* Handle packets as part of this loop, not in the IRQ handler */
162  lpc_enetif_input(&lpc_netif);
163 
164  /* lpc_rx_queue will re-qeueu receive buffers. This normally occurs
165  automatically, but in systems were memory is constrained, pbufs
166  may not always be able to get allocated, so this function can be
167  optionally enabled to re-queue receive buffers. */
168 #if 0
169  while (lpc_rx_queue(&lpc_netif)) {}
170 #endif
171 
172  /* Free TX buffers that are done sending */
173  lpc_tx_reclaim(&lpc_netif);
174 
175  /* LWIP timers - ARP, DHCP, TCP, etc. */
176  sys_check_timeouts();
177 
178  /* Call the PHY status update state machine once in a while
179  to keep the link status up-to-date */
180  physts = lpcPHYStsPoll();
181 
182  /* Only check for connection state when the PHY status has changed */
183  if (physts & PHY_LINK_CHANGED) {
184  if (physts & PHY_LINK_CONNECTED) {
185  Board_LED_Set(0, true);
186  prt_ip = 0;
187 
188  /* Set interface speed and duplex */
189  if (physts & PHY_LINK_SPEED100) {
190  Chip_ENET_Set100Mbps(LPC_ETHERNET);
191  NETIF_INIT_SNMP(&lpc_netif, snmp_ifType_ethernet_csmacd, 100000000);
192  }
193  else {
194  Chip_ENET_Set10Mbps(LPC_ETHERNET);
195  NETIF_INIT_SNMP(&lpc_netif, snmp_ifType_ethernet_csmacd, 10000000);
196  }
197  if (physts & PHY_LINK_FULLDUPLX) {
198  Chip_ENET_SetFullDuplex(LPC_ETHERNET);
199  }
200  else {
201  Chip_ENET_SetHalfDuplex(LPC_ETHERNET);
202  }
203 
204  netif_set_link_up(&lpc_netif);
205  }
206  else {
207  Board_LED_Set(0, false);
208  netif_set_link_down(&lpc_netif);
209  }
210 
211  DEBUGOUT("Link connect status: %d\r\n", ((physts & PHY_LINK_CONNECTED) != 0));
212  }
213 
214  /* Print IP address info */
215  if (!prt_ip) {
216  if (lpc_netif.ip_addr.addr) {
217  static char tmp_buff[16];
218  DEBUGOUT("IP_ADDR : %s\r\n", ipaddr_ntoa_r((const ip_addr_t *) &lpc_netif.ip_addr, tmp_buff, 16));
219  DEBUGOUT("NET_MASK : %s\r\n", ipaddr_ntoa_r((const ip_addr_t *) &lpc_netif.netmask, tmp_buff, 16));
220  DEBUGOUT("GATEWAY_IP : %s\r\n", ipaddr_ntoa_r((const ip_addr_t *) &lpc_netif.gw, tmp_buff, 16));
221  prt_ip = 1;
222  }
223  }
224  }
225 
226  /* Never returns, for warning only */
227  return 0;
228 }
