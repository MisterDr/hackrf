53 /* assign a movable pin function to a physical pin */
54 void Chip_SWM_MovablePinAssign(CHIP_SWM_PIN_MOVABLE_T movable, CHIP_PINx_T pin)
55 {
56  uint32_t temp;
57  temp = LPC_SWM->PINASSIGN[PINASSIGN_IDX(movable)] & (~(0xFF << PINSHIFT(movable)));
58  LPC_SWM->PINASSIGN[PINASSIGN_IDX(movable)] = temp | (pin << PINSHIFT(movable));
59 }

61 /* true enables, false disables a Switch Matrix fixed-pin Function */
62 void Chip_SWM_FixedPinEnable(CHIP_SWM_PIN_FIXED_T pin, bool enable)
63 {
64  if (enable) { /* clear the bit */
65  LPC_SWM->PINENABLE0 &= ~(1 << pin);
66  }
67  else { /* set the bit */
68  LPC_SWM->PINENABLE0 |= (1 << pin);
69  }
70 }

72 /* returns true if pin function is enabled, false if disabled */
73 bool Chip_SWM_IsEnabled(CHIP_SWM_PIN_FIXED_T pin)
74 {
75  if (LPC_SWM->PINENABLE0 & (1 << pin)) {
76  return false; /* the bit is set, has not been enabled */
77  }
78  return true;/* the bit is not set, therefore is enabled */
79 }
