54 /* Put some of the peripheral in sleep mode */
55 void Chip_PMU_Sleep(LPC_PMU_T *pPMU, CHIP_PMU_MCUPOWER_T SleepMode)
56 {
57  /* Currently powered devices will not be powered down */
58  Chip_SYSCTL_SetWakeup(Chip_SYSCTL_GetPowerStates());
59 
60  /* Disable powerdown for WDT and brown-out detector */
61  Chip_SYSCTL_SetDeepSleepPD(~(SYSCTL_DEEPSLP_BOD_PD | SYSCTL_DEEPSLP_WDTOSC_PD));
62 
63  switch ( SleepMode ) {
64  case PMU_MCU_DEEP_SLEEP:
65  SCB->SCR |= M0SCB_SLEEPDEEP;
66  pPMU->DPDCTRL |= PMU_DPDCTRL_LPOSCEN;
67  pPMU->PCON = PMU_PCON_PM_DPD;
68  break;
69 
70  case PMU_MCU_POWER_DOWN:
71  SCB->SCR |= M0SCB_SLEEPDEEP;
72  pPMU->DPDCTRL |= PMU_DPDCTRL_LPOSCEN;
73  pPMU->PCON = PMU_PCON_PM_PWD;
74  break;
75 
76  case PMU_MCU_DEEP_PWRDOWN:
77  pPMU->GPREG0 = 0x12345678;
78  pPMU->GPREG1 = 0x87654321;
79  pPMU->GPREG2 = 0x56781234;
80  pPMU->GPREG3 = 0x43218765;
81  SCB->SCR |= M0SCB_SLEEPDEEP;
82  pPMU->DPDCTRL |= PMU_DPDCTRL_LPOSCDPDEN;
83  pPMU->PCON = PMU_PCON_PM_DPWD;
84  break;
85 
86  case PMU_MCU_SLEEP:
87  pPMU->DPDCTRL |= PMU_DPDCTRL_LPOSCEN;
88 
89  default:
90  break;
91  }
92  __WFI();/* Enter to sleep mode */
93 }
