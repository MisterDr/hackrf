104 void MRT_IRQHandler(void)
105 {
106  uint32_t idx;
107  uint32_t int_pend;
108 
109  int_pend = Chip_MRT_GetIntPending();
110  switch (int_pend) {
111  case MRT0_INTFLAG:
112  idx = 0;
113  break;
114 
115  case MRT1_INTFLAG:
116  idx = 1;
117  break;
118 
119  case MRT2_INTFLAG:
120  idx = 2;
121  break;
122 
123  case MRT3_INTFLAG:
124  idx = 3;
125  break;
126 
127  default:/* should never get here */
128  while (1) {
129  Board_LED_Toggle(0);
130  }
131  }
132 
133  Chip_MRT_IntClear(&LPC_MRT->CHANNEL[idx]);
134  Board_LED_Toggle(2);
135  __SEV();
136 }

142 int main(void)
143 {
144  /* a pointer to a timer */
145  LPC_MRT_CHANNEL_T *timer;
146 
147  /* Generic Initialization */
148  Board_Init();
149 
150  /* MRT Initialization */
151  Chip_MRT_Init();
152 
153  /* Enable SysTick Timer */
154  SysTick_Config(SystemCoreClock / TICKRATE_HZ);
155 
156  DEBUGSTR("LPC8xx MRT Example \r\n");
157 
158  /* Enable the interrupt for the MRT */
159  NVIC_EnableIRQ(MRT_IRQn);
160 
161  while (1) {
162  /* Check if IDLE channel available in MRT */
163  if (Chip_MRT_GetIdleChannel() != LPC_MRT_NO_IDLE_CHANNEL) {
164  Board_LED_Set(0, false);
165 
166  /* Get channel structure of IDLE channel */
167  timer = (LPC_MRT_CHANNEL_T *) (LPC_MRT_BASE + Chip_MRT_GetIdleChannel());
168 
169  /* Generate new pseudo-random number */
170  RAND();
171 
172  /* Configure timer in single-shot mode */
173  Chip_MRT_SetMode(timer, MRT_MODE_ONESHOT);
174 
175  /* Set some random amount of time interval */
176  Chip_MRT_SetInterval(timer, rng / 100);
177 
178  /* Enable the channel */
179  Chip_MRT_SetEnabled(timer);
180  }
181  else {
182  /* indicate that no timer channel is available */
183  Board_LED_Set(0, true);
184  }
185  __WFE();
186  }
187 
188  return 0;
189 }
