113 /* Set the pin mode (pull-up/pull-down). */
114 void Chip_IOCON_PinSetMode(LPC_IOCON_T *pIOCON, CHIP_PINx_T pin, CHIP_PIN_MODE_T mode)
115 {
116  switch (pin) {
117  case PIO10: /* No mode for the I2C pins */
118  case PIO11:
119  break;
120 
121  default:
122  PIN_MODE_CLR(pin);
123  PIN_MODE_SET(pin, mode);
124  break;
125  }
126 }
