60 /* Set baud rate for UART */
61 void Chip_UART_SetBaudRate(LPC_USART_T *pUART, uint32_t baudrate)
62 {
63  uint32_t err, uart_fra_multiplier, baudRateGenerator;
64  uint32_t systemCoreClock = Chip_Clock_GetMainClockRate();
65 
66  /* Calculate baudrate generator value */
67  baudRateGenerator = systemCoreClock / (16 * baudrate);
68  err = systemCoreClock - baudRateGenerator * 16 * baudrate;
69  uart_fra_multiplier = (err * 0xFF) / (baudRateGenerator * 16 * baudrate);
70  IP_UART_SetBaudRate(pUART, baudRateGenerator - 1); /*baud rate */
71  Chip_SYSCTL_SetUSARTFRGDivider(0xFF); /* value 0xFF is always used */
72  Chip_SYSCTL_SetUSARTFRGMultiplier(uart_fra_multiplier);
73 }

75 /* Transmit a 8 bits data */
76 Status Chip_UART_SendByte(LPC_USART_T *pUART, uint8_t data)
77 {
78  if (IP_UART_GetStatus(pUART) & (TXRDY)) {
79  IP_UART_Transmit(pUART, data);
80  return SUCCESS;
81  }
82  return ERROR;
83 }

85 /* Receive a 8 bits data */

87 {
88  if (IP_UART_GetStatus(pUART) & (RXRDY)) {
89  *Data = IP_UART_Receive(pUART);
